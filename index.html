⚠️ The tradedge_trades table — run this SQL once in Supabase → SQL Editor: 
-- ① Fresh install: create table
CREATE TABLE IF NOT EXISTS tradedge_trades (
  device_id   TEXT PRIMARY KEY,
  trades_json TEXT,
  ledger_json TEXT,
  cfg_json    TEXT,
  fund_json   TEXT,
  alerts_json TEXT,
  prefs_json  TEXT,
  updated_at  TIMESTAMPTZ DEFAULT now()
);
ALTER TABLE tradedge_trades ENABLE ROW LEVEL SECURITY;
CREATE POLICY "allow_all" ON tradedge_trades
  FOR ALL USING (true) WITH CHECK (true);

-- ② Existing table: add missing columns (run if ledger not syncing)
ALTER TABLE tradedge_trades ADD COLUMN IF NOT EXISTS ledger_json TEXT;
ALTER TABLE tradedge_trades ADD COLUMN IF NOT EXISTS cfg_json    TEXT;
ALTER TABLE tradedge_trades ADD COLUMN IF NOT EXISTS fund_json   TEXT;
ALTER TABLE tradedge_trades ADD COLUMN IF NOT EXISTS alerts_json TEXT;
